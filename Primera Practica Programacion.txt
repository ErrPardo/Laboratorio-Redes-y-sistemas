Ejercicio A
	var
		decimal: peso, estatura, IMC
	Inicio
		Escribir ("¿ Cuanto pesas en kg y mides en centímetros ?"

			estatura<-estatura/100
			IMC<-peso/(estatura)**2
		Escribir ("índice de masa corporal:",IMC)
	Fin
	
Ejercicio B
	var	
		decimal:radio,altura,volumen
		const:pi<-3.1415926535
	Inicio
	volumen<-0
		Escribir ("¿ Cuanta altura y radio quieres que tenga el volumen del cilíndro ?")
		Leer (altura,radio)
			volumen<-pi*altura*radio**2
		Escribir ("Este es el volumen de tu cilindro:",volumen)
	Fin

Ejercicio D
